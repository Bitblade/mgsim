239  /*
240  A register pair is an output and input register on different cores that
241  are directly connected to each other. It contains a single process that
242  moves data from the output on one core to the input register on the
243  other core.
244 
245  The network class should have a process to be sensitive on the input
246  register.
247  */
248  template <typename T>
249  class RegisterPair : public Object
250  {
251  private:
252  Register<T>* remote;
253  Process p_Transfer;
254 
255  public:
256  Register<T> out;
257  Register<T> in;
258 
260  Result DoTransfer()
261  {
262  assert(!out.Empty());
263  assert(remote != NULL);
264  if (!remote->Write(out.Read()))
265  {
266  return FAILED;
267  }
268  out.Clear();
269  return SUCCESS;
270  }
271 
273  void Initialize(RegisterPair<T>& dest)
274  {
275  assert(remote == NULL);
276  remote = &dest.in;
277  dest.in.AddProcess(p_Transfer);
278  p_Transfer.SetStorageTraces(dest.in);
279  }
280 
281  RegisterPair(Object& parent, const std::string& name)
282  : Object(name, parent),
283  remote(NULL),
284  p_Transfer(*this, "transfer", delegate::create<RegisterPair, &RegisterPair::DoTransfer>(*this)),
285  out(parent, name + ".out"),
286  in (parent, name + ".in")
287  {

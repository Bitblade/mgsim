207 template <typename I>
208  class Structure : public IStructure
209 {
210  typedef std::map<WritePort<I>*, int> PriorityMap;
211  typedef std::set<ArbitratedWritePort<I>*> ArbitratedWritePortList;
212  typedef std::set<WritePort<I>*> WritePortList;
213 
214  void OnArbitrate()
215  {
216  typedef std::vector<WritePort<I>*> WritePortMap;
217  typedef std::map<I,WritePortMap> RequestPortMap;
218 
219  // Tell each port to arbitrate
220  ArbitrateReadPorts();
221  for (auto i : m_arbitratedWritePorts)
222  {
223  i->Arbitrate();
224  }
225 
226  // Get the final requests from all ports
227  RequestPortMap requests;
228  for (auto i : m_writePorts)
229  {
230  const I* index = i->GetIndex();
231  if (index != NULL)
232  {
233  requests[*index].push_back(i);
234  }
235  }
236 
237  // Arbitrate between the ports for each request
238  for (auto& i : requests)
239  {
240  WritePort<I>* port = NULL;
241  int highest = std::numeric_limits<int>::max();
242 
243  for (auto j : i.second)
244  {
245  auto priority = m_priorities.find(j);
246  if (priority != m_priorities.end() && priority->second < highest)
247  {
248  highest = priority->second;
249  port = j;
250  }
251  }
252 
253  for (auto j : i.second)
254  {
255  j->Notify( port == j );
256  }
257  }
258  }
259 
260  WritePortList m_writePorts;
261  ArbitratedWritePortList m_arbitratedWritePorts;
262  PriorityMap m_priorities;
263 
264 public:
265  Structure(const std::string& name, Object& parent, Clock& clock)
266  : Object(name, parent, clock),
267  IStructure(name, parent, clock),
268  m_writePorts(),
269  m_arbitratedWritePorts(),
270  m_priorities()
271  {}
